Sure, here is a step-by-step guide on how to scan your movies and shows folders, use the scanned data to send to an API to get IMDB info, and then show that info in your EJS template:

1. Scan your movies and shows folders

You can use the following Node.js code to scan your movies and shows folders:

JavaScript
const fs = require('fs');

function scanFolder(folderPath) {
  const files = fs.readdirSync(folderPath);
  const movies = [];
  const shows = [];

  for (const file of files) {
    const filePath = `${folderPath}/${file}`;
    const stats = fs.statSync(filePath);

    if (stats.isFile()) {
      const extension = file.split('.').pop();

      if (extension === 'mp4' || extension === 'mkv') {
        movies.push(filePath);
      } else if (extension === 'avi' || extension === 'wmv') {
        shows.push(filePath);
      }
    }
  }

  return { movies, shows };
}

const movieAndShowPaths = scanFolder('/path/to/movies-and-shows');

const axios = require('axios');

function getIMDBInfo(movieOrShowPath) {
  const imdbAPIKey = 'YOUR_IMDB_API_KEY';
  const imdbURL = `https://api.themoviedb.org/3/search/movie?api_key=${imdbAPIKey}&query=${movieOrShowPath}`;

  return axios.get(imdbURL)
    .then(response => response.data.results[0]);
}
